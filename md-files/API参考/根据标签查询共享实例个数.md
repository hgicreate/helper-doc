        "action": "count",

            "values": ["value2"]
        }, {
            "key": "key2",
            "values": []
        }],
        "tags_any": [{
            "key": "key3",
            "values": ["value3"]
        }, {
            "key": "key4",
            "values": []
        }],
        "not_tags": [{
            "key": "key5",
            "values": ["value5"]
        }, {

            "values": []
        }],
        "not_tags_any": [{

            "values": ["value7", "value8"]

            "values": []
